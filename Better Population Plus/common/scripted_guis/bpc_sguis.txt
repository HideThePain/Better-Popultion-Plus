        exists = global_var:special_congenital_protection
    }
    effect = {
        if = {
            limit = {
                exists = global_var:special_congenital_protection
            }
            remove_global_variable = special_congenital_protection
        }
        else = {
            set_global_variable = special_congenital_protection
        }
    }
}

    is_valid = {
        global_var:base_death_chance < 100
    }
    effect = {
        change_global_variable = {
            name = base_death_chance
            add = 5
        }
    }
}
